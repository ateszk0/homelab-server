# 1. Telepítés
sh <(curl -sSf https://downloads.nordcdn.com/apps/linux/install.sh)

# 2. Belépés Tokennel
nordvpn login --token TE_TOKENED

# 3. Meshnet bekapcsolása
nordvpn set meshnet on
nordvpn set technology nordlynx

# 4. Routing engedélyezése (Hogy telefonról elérd a belső hálót)
# Listázd ki az eszközöket:
nordvpn meshnet peer list
# Engedélyezd a telefonodat (cseréld a nevet!):
nordvpn meshnet peer routing allow telefon-neve.nord
nordvpn meshnet peer local allow telefon-neve.nord
