# 1. SMART eszközök telepítése
apt update && apt install smartmontools -y

# 2. Kézi teszt (Diagnosztika)
smartctl -a /dev/nvme0n1       # Belső NVMe
smartctl -d sat -a /dev/sda    # Külső USB SSD (SAT módban)

# 3. Scrutiny Config mappa létrehozása
mkdir -p /root/monitor/scrutiny-config
nano /root/monitor/scrutiny-config/collector.yaml
# (Tartalom lejjebb)

# 4. Kényszerített adatgyűjtés (Ha nem frissülne)
docker exec scrutiny /opt/scrutiny/bin/scrutiny-collector-metrics run
